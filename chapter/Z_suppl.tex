\begin{algorithm}
  \caption{Training Algorithm}
  \label{alg:training_algorithm}

    $x_0\sim q(x_0)$ \\
    $y\sim \mathrm{Uniform}(0,1)$ \\
    Take gradient descent step on
    \[
      \nabla_\theta \| y - f_\theta(x_0) \|^2
    \]
  }
